76 // Convert local address to global address.
77 static inline uint32_t tomp_makeAddressGlobal(uint32_t core_idx,
78  uint32_t address)
79 {
80 #if defined(TI_AM572)
81  if (address < 0x01000000)
82  {
83  if (core_idx == 0)
84  {
85  // dsp1_l2_sram 0x4080_0000 - 0x4084_7fff
86  address = 0x40800000 | (address & 0x000FFFFF);
87  }
88  else if (core_idx == 1)
89  {
90  // dsp2_l2_sram 0x4100_0000 - 0x4104_7fff
91  address = 0x41000000 | (address & 0x000FFFFF);
92  }
93  }
94 #else
95  if (address < 0x01000000)
96  {
97  // Mask: 0x00****** => 0x1"core_idx"******
98  address = (0x10 + core_idx) << 24 | address;
99  }
100 #endif
101 
102  return address;
103 }

106 static inline void tomp_mfence(void)
107 {
108  // In order to accurately trace mfence instructions, a nop and mark
109  // instruction must follow each mfence instruction
110  asm (" mfence ");
111  asm(" nop ");
112  asm(" mark 0");
113  asm (" mfence "); // 2nd MFENCE as per Keystone I FAE Alert
114  asm(" nop ");
115  asm(" mark 0");
116 
117  // sprz332b.pdf, Advisory 24 - Require 16 NOPs after MFENCE after certain
118  // cache coherency operations
119  asm(" nop 9");
120  asm(" nop 7");
121 }

274 static inline void tomp_waitAtBarrier(tomp_Barrier *barrier)
275 {
276  // Complete any pending tasks. Implicit and explicit barriers are task
277  // synchronization points.
278  tomp_completePendingTasks();
279 
280  // For re-use reason, the barrier contains a sense variation
281  unsigned int mysense = !(barrier->sense);
282 
283  // Pass only if it's the last thread
284  if (tomp_decrementAtomic(&barrier->count) == 1)
285  {
286  // Reset of the barrier
287  barrier->count = barrier->size;
288  barrier->sense = !(barrier->sense);
289  }
290  else
291  {
292  // A modification of the sense represents the end of the barrier
293  // Check for pending tasks and execute them when waiting
294  while (mysense != barrier->sense)
295  tomp_completePendingTasks();
296  }
297 }

303 static inline tomp_waitAtBarrierWithFlush(tomp_Barrier *barrier)
304 {
305  // Wait for all writes to land before exiting the barrier. Will ensure
306  // that inadvertent reads of shared variables by worker cores when
307  // executing runtime code will pull in updated values into their
308  // caches.
309  tomp_mfence();
310 
311  // Wait for other threads on the team to reach the barrier. Will also
312  // execute any pending tasks
313  tomp_waitAtBarrier(barrier);
314 
315  // OMP v3.0, 2.8.6, p75: Implied flush at exit from parallel region
316  // Flush AFTER all threads have reached the barrier. This handles a
317  // corner case where a shared variable on the stack is on the same
318  // cache line as the location of return address on the stack. The read
319  // of the return address during function epilog also reads in the
320  // shared variable into the cache and marks it as valid. Because of
321  // this inadvertent read of the shared variable, we need to ensure that
322  // writes to the shared variable from all threads have landed before
323  // we perform the flush.
324  tomp_cacheWbInvAll();
325 }

354 static inline bool tomp_isPtrCached(void *ptr)
355 {
356  volatile uint32_t *MAR = (volatile uint32_t *)MAR_ADDRESS_BASE;
357 
358  // Compute the MAR index for the pointer
359  int idx = (uint32_t)ptr >> 24;
360 
361  // Need to special case L2SRAM as this space could be mapped to a
362  // global address space by ORing with (DNUM + 16) << 24.
363  // The MAR indices corresponding to this global space are invalid.
364  // Assumption: any variable in L2SRAM scratch is cached in L1
365  if ((idx & (~(DNUM + 16))) == 0)
366  return true;
367 
368  // if bit 0 (PC) of the MAR is 1, the address has caching enabled
369  if (MAR[idx] & 0x1)
370  return true;
371 
372  return false;
373 }
