37 typedef struct { void *ptr; volatile unsigned int lock; } gomp_ptrlock_t;
38 
39 static inline void gomp_ptrlock_init (gomp_ptrlock_t *ptrlock, void *ptr)
40 {
41  ptrlock->ptr = ptr;
42  ptrlock->lock = TOMP_WS_HW_SEM_IDX;
43 }
44 
45 static inline void *gomp_ptrlock_get (gomp_ptrlock_t *ptrlock)
46 {
47  if (ptrlock->ptr != NULL)
48  return ptrlock->ptr;
49 
50  /* First thread to hit this point from gomp_loop_*_start acquires the lock
51  * All remaining threads will sit and
52  * spin here waiting for the lock to be released by the first thread
53  * via gomp_ptrlock_set
54  */
55  tomp_mutex_lock (ptrlock->lock);
56  if (ptrlock->ptr != NULL)
57  {
58  tomp_mutex_unlock (ptrlock->lock);
59  return ptrlock->ptr;
60  }
61 
62  return NULL;
63 }
64 
65 static inline void gomp_ptrlock_set (gomp_ptrlock_t *ptrlock, void *ptr)
66 {
67  ptrlock->ptr = ptr;
68  tomp_mutex_unlock (ptrlock->lock);
69 }
70 
71 static inline void gomp_ptrlock_destroy (gomp_ptrlock_t *ptrlock)
72 {
73  /* Nothing to do here */
74 }

76 #endif /* GOMP_PTRLOCK_H */
