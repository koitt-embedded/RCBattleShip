91 static inline void *queuePop(sw_queue *Q)
92 {
93  void *item = NULL;
94 
95  if (Q->count != 0)
96  {
97  queue_lock_acquire(Q);
98 
99  if (Q->count != 0) // not empty
100  {
101  item = Q->events[Q->head];
102  Q->head = (Q->head + 1) % TOMP_EVENT_NUM;
103  Q->count -= 1;
104  if ((int)Q->count < 0)
105  tomp_logError(tomp_ErrorKind_QUEUE, __FILE__, __LINE__);
106  }
107 
108  queue_lock_release(Q);
109  }
110 
111  return item;
112 }

114 static inline queuePushDesc(sw_queue *Q, void *event)
115 {
116  uint32_t tail;
117 
118  queue_lock_acquire(Q);
119 
120  if (Q->count == TOMP_EVENT_NUM)
121  tomp_logError(tomp_ErrorKind_QUEUE, __FILE__, __LINE__);
122  else
123  {
124  tail = (Q->head + Q->count) % TOMP_EVENT_NUM;
125  Q->events[tail] = event;
126  Q->count += 1;
127  }
128 
129  queue_lock_release(Q);
130 }

132 static inline void queuePush(sw_queue *Q, void *event)
133 {
134  queue_lock_acquire(Q);
135 
136  if (Q->count == TOMP_EVENT_NUM)
137  tomp_logError(tomp_ErrorKind_QUEUE, __FILE__, __LINE__);
138  else
139  {
140  if (Q->head == 0) Q->head = TOMP_EVENT_NUM-1;
141  else Q->head = (Q->head - 1) % TOMP_EVENT_NUM;
142  Q->events[Q->head] = event;
143  Q->count++;
144  }
145 
146  queue_lock_release(Q);
147 }
